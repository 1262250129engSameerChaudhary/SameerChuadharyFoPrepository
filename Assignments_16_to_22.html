ASSIGNMENT NO. 16
Partition List into Even and Odd Numbers
Accept a list of N integers and partition it into two sublists containing even and odd numbers respectively.
C Code
#include <stdio.h>

int main() {
    int n, i;
    printf("Enter number of elements: ");
    scanf("%d", &n);

    int arr[n], even[n], odd[n];
    int ec = 0, oc = 0;

    printf("Enter %d integers: ", n);
    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    for (i = 0; i < n; i++) {
        if (arr[i] % 2 == 0)
            even[ec++] = arr[i];   // Even number
        else
            odd[oc++] = arr[i];    // Odd number
    }

    printf("\nEven numbers are: ");
    for (i = 0; i < ec; i++)
        printf("%d ", even[i]);

    printf("\nOdd numbers are: ");
    for (i = 0; i < oc; i++)
        printf("%d ", odd[i]);

    printf("\n");
    return 0;
}
Sample I/O
Input:
Enter number of elements: 5
Enter 5 integers: 10 7 4 9 12

Output:
Even numbers are: 10 4 12
Odd numbers are: 7 9
FAQs
  • The % (modulus) operator returns the remainder when dividing by 2; if 0 the number is even.
  • Arrays allow storing multiple values of the same type so N integers can be processed together in a loop.
  • The modulus operator classifies each element: arr[i]%2==0 → even, else → odd.

ASSIGNMENT NO. 17
Sum of Sine Series (Taylor Expansion)
Accept the value of x and the number of terms, and compute the sum of the sine series using the Taylor expansion.
C Code
#include <stdio.h>
#include <math.h>

int main() {
    double x, sum = 0.0, term;
    int n, i, j;
    long long fact;

    printf("Enter value of x (in radians): ");
    scanf("%lf", &x);
    printf("Enter number of terms: ");
    scanf("%d", &n);

    for (i = 0; i < n; i++) {
        int power = 2 * i + 1;   // 1, 3, 5, 7, ...

        // Compute (2i+1)! using a loop
        fact = 1;
        for (j = 1; j <= power; j++)
            fact *= j;

        term = pow(x, power) / (double)fact;

        if (i % 2 == 0)
            sum += term;   // Positive term
        else
            sum -= term;   // Negative term
    }

    printf("Sum of sine series = %lf\n", sum);
    return 0;
}
/* Compile: gcc sine.c -lm -o sine */
Sample I/O
Input:
Enter value of x (in radians): 1
Enter number of terms: 5

Output:

Formula Used
Taylor Series
sin(x) = x − x³/3! + x⁵/5! − x⁷/7! + ...

i-th Term
term_i = (−1)^i · x^(2i+1) / (2i+1)!

ASSIGNMENT NO. 18
String Operations in C
Perform: length, reversal, equality check, palindrome check, and substring check on user-input strings.
C Code
#include <stdio.h>
#include <string.h>

int main() {
    char s1[100], s2[100], rev[100];
    int len, i;

    printf("Enter a string: ");
    scanf("%s", s1);

    /* i) Length of string */
    len = strlen(s1);
    printf("Length of string = %d\n", len);

    /* ii) String reversal */
    for (i = 0; i < len; i++)
        rev[i] = s1[len - 1 - i];
    rev[len] = '\0';
    printf("Reversed string = %s\n", rev);

    /* iv) Palindrome check */
    if (strcmp(s1, rev) == 0)
        printf("The string is a Palindrome\n");
    else
        printf("The string is NOT a Palindrome\n");

    /* iii) Equality check with another string */
    printf("Enter another string for equality check: ");
    scanf("%s", s2);

    if (strcmp(s1, s2) == 0)
        printf("Strings are equal\n");
    else
        printf("Strings are not equal\n");

    /* v) Substring check */
    if (strstr(s1, s2) != NULL)
        printf("Substring found in main string\n");
    else
        printf("Substring NOT found in main string\n");

    return 0;
}

Employee Structure in C
Create a structure EMPLOYEE and perform operations: total count, male/female count, salary > 10,000, designation filter.
C Code
#include <stdio.h>
#include <string.h>

struct EMPLOYEE {
    char name[50];
    char designation[50];
    char gender[10];
    char doj[15];   // Date of Joining (DD-MM-YYYY)
    float  salary;
};

int main() {
    int n, i, male = 0, female = 0;

    printf("Enter number of employees: ");
    scanf("%d", &n);

    struct EMPLOYEE emp[n];

    for (i = 0; i < n; i++) {
        printf("\nEmployee %d:\n", i + 1);
        printf("  Name        : "); scanf("%s", emp[i].name);
        printf("  Designation : "); scanf("%s", emp[i].designation);
        printf("  Gender      : "); scanf("%s", emp[i].gender);
        printf("  Date of Joining: "); scanf("%s", emp[i].doj);
        printf("  Salary      : "); scanf("%f", &emp[i].salary);

        if (strcmp(emp[i].gender, "Male") == 0)   male++;
        else                                          female++;
    }

    /* a) Total employees */
    printf("\nTotal number of employees = %d\n", n);

    /* b) Male/Female count */
    printf("Male employees   = %d\n", male);
    printf("Female employees = %d\n", female);

    /* c) Salary > 10000 */
    printf("\nEmployees with salary > 10000:\n");
    for (i = 0; i < n; i++)
        if (emp[i].salary > 10000)
            printf("  %s (%.2f)\n", emp[i].name, emp[i].salary);

    /* d) Designation = "Asst Manager" */
    printf("\nEmployees with designation 'AsstManager':\n");
    for (i = 0; i < n; i++)
        if (strcmp(emp[i].designation, "AsstManager") == 0)
            printf("  %s\n", emp[i].name);

    return 0;
}

ASSIGNMENT NO. 20
Swap Two Numbers — With & Without Pointers
Demonstrate Call by Value (no effect on original) vs Call by Reference (pointers) for swapping two integers.
C Code
#include <stdio.h>

/* Call by Value — changes do NOT reflect in main */
void swapWithoutPointer(int a, int b) {
    int temp = a;
    a = b;
    b = temp;
    printf("Inside function (no pointer): a = %d, b = %d\n", a, b);
}

/* Call by Reference — directly modifies original variables */
void swapWithPointer(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

int main() {
    int a, b;
    printf("Enter two numbers: ");
    scanf("%d %d", &a, &b);

    /* Without pointers */
    printf("\n--- Without Pointers (Call by Value) ---\n");
    swapWithoutPointer(a, b);
    printf("In main after call: a = %d, b = %d\n", a, b);
    // a and b are unchanged in main

    /* With pointers */
    printf("\n--- With Pointers (Call by Reference) ---\n");
    swapWithPointer(&a, &b);
    printf("After swapping with pointers: a = %d, b = %d\n", a, b);

    return 0;
}
Sample I/O
Enter two numbers: 10 20

--- Without Pointers ---
Inside function: a=20, b=10
In main after call: a=10, b=20

--- With Pointers ---
After swapping: a=20, b=10
Concept Comparison
Call by Value
Copy is passed → original unchanged in main
Call by Reference
Address passed → original is modified
Pointer dereference
*a accesses value at address a
Address-of
&a gives the memory address of a

File Copy using File Handling
Read characters from source.txt one by one using fgetc() and write them to destination.txt using fputc() until EOF.
C Code
#include <stdio.h>

int main() {
    FILE *src, *dest;
    int ch;   /* Must be int to correctly detect EOF (-1) */
    char srcFile[]  = "source.txt";
    char destFile[] = "destination.txt";

    /* Open source file in READ mode */
    src = fopen(srcFile, "r");
    if (src == NULL) {
        printf("Error: Could not open source file '%s'\n", srcFile);
        return 1;
    }

    /* Open destination file in WRITE mode */
    dest = fopen(destFile, "w");
    if (dest == NULL) {
        printf("Error: Could not create destination file\n");
        fclose(src);
        return 1;
    }

    /* Copy characters until EOF */
    while ((ch = fgetc(src)) != EOF) {
        fputc(ch, dest);
    }

    fclose(src);
    fclose(dest);

    printf("File copied successfully.\n");
    printf("Contents copied from '%s' to '%s'\n", srcFile, destFile);

    return 0;
}

Menu-Driven String Operations (Built-in Functions)
A menu-driven C program using strlen, strcpy, strcat, strcmp, and manual reversal from string.h.
C Code
#include <stdio.h>
#include <string.h>

int main() {
    char s1[100], s2[100];
    int choice, i;

    printf("Enter first string : ");  scanf("%s", s1);
    printf("Enter second string: ");  scanf("%s", s2);

    do {
        printf("\n===== String Operations Menu =====\n");
        printf("1. Length of first string (strlen)\n");
        printf("2. Copy s1 to a new string (strcpy)\n");
        printf("3. Concatenate s1 + s2 (strcat)\n");
        printf("4. Compare s1 and s2 (strcmp)\n");
        printf("5. Reverse of s1\n");
        printf("6. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                printf("Length of s1 = %d\n", (int)strlen(s1));
                break;

            case 2: {
                char copy[100];
                strcpy(copy, s1);
                printf("Copied string: %s\n", copy);
                break;
            }

            case 3: {
                char cat[200];
                strcpy(cat, s1);
                strcat(cat, s2);
                printf("Concatenated string: %s\n", cat);
                break;
            }

            case 4:
                if (strcmp(s1, s2) == 0)
                    printf("Strings are EQUAL\n");
                else
                    printf("Strings are NOT equal\n");
                break;

            case 5: {
                int len = strlen(s1);
                char rev[100];
                for (i = 0; i < len; i++)
                    rev[i] = s1[len - 1 - i];
                rev[len] = '\0';
                printf("Reversed string: %s\n", rev);
                break;
            }

            case 6:
                printf("Exiting program...\n");
                break;

            default:
                printf("Invalid choice! Try again.\n");
        }
    } while (choice != 6);

    return 0;
}
